#随机弹出list的内容，每次一个
from random import randint
def randGen(aList):
    while len(aList)>0:
        yield aList.pop(randint(0,len(aList)-1))

for item in randGen(['rock', 'paper', 'scissors']):
    print item


def counter(start_at=0):
    count = start_at
    while True:
        val = (yield count) 
        if val is not None:
            count = val
        else:
            count+=1

count = counter(5)
print count.next() #5
print count.next() #6
count.send(10) #发送10
print count.next() #11
count.close() #关闭
